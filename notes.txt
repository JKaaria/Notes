Sub ArrangeTablesOnOverview()
    Dim wsOverview As Worksheet
    Dim lastTableBottom As Long
    Dim table As ListObject

    ' Check if "Overview" sheet exists, if yes, use it
    On Error Resume Next
    Set wsOverview = ThisWorkbook.Sheets("Overview")
    On Error GoTo 0

    If wsOverview Is Nothing Then
        MsgBox "Sheet 'Overview' not found!", vbExclamation
        Exit Sub
    End If

    ' Clear any existing content below the last table
    lastTableBottom = wsOverview.ListObjects(wsOverview.ListObjects.Count).HeaderRowRange.Row + _
                      wsOverview.ListObjects(wsOverview.ListObjects.Count).ListRows.Count + 1

    If lastTableBottom > wsOverview.Rows.Count Then
        lastTableBottom = wsOverview.Rows.Count
    End If

    wsOverview.Rows(lastTableBottom & ":" & wsOverview.Rows.Count).Clear

    ' Loop through all tables in the "Overview" sheet
    For Each table In wsOverview.ListObjects
        ' Move the table below the last table with a small gap
        table.HeaderRowRange.Row = lastTableBottom + 2
        lastTableBottom = table.HeaderRowRange.Row + table.ListRows.Count + 1
    Next table

    ' Format the "Overview" sheet
    wsOverview.Columns.AutoFit

    MsgBox "Tables on Overview arranged successfully!", vbInformation
End Sub
