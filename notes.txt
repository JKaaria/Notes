Sub InsertLookupInColumnT()
    Dim cashFailsSheet As Worksheet
    Dim inventorySheet As Worksheet
    Dim cashFailsLastRow As Long
    Dim inventoryLastRow As Long
    Dim i As Long
    
    ' Set reference to active sheet (FI_Cash_Fails)
    Set cashFailsSheet = ActiveSheet
    
    ' Set reference to the inventory sheet (FI_FISS_Inventory)
    On Error Resume Next
    Set inventorySheet = Worksheets("FI_FISS_Inventory")
    On Error GoTo 0
    
    ' Check if the inventory sheet exists
    If inventorySheet Is Nothing Then
        MsgBox "FI_FISS_Inventory sheet not found.", vbExclamation
        Exit Sub
    End If
    
    ' Find the last row in both sheets
    cashFailsLastRow = cashFailsSheet.Cells(cashFailsSheet.Rows.Count, "U").End(xlUp).Row
    inventoryLastRow = inventorySheet.Cells(inventorySheet.Rows.Count, "D").End(xlUp).Row
    
    ' Loop through each row in FI_Cash_Fails
    For i = 2 To cashFailsLastRow ' Assuming header is in the first row
        ' Use the Match function to find the corresponding value in FI_FISS_Inventory
        Dim matchResult As Variant
        matchResult = Application.Match(cashFailsSheet.Cells(i, "U").Value, inventorySheet.Range("D2:D" & inventoryLastRow), 0)
        
        ' Check if a match is found
        If Not IsError(matchResult) Then
            ' If a match is found, insert the value from FI_FISS_Inventory column H into FI_Cash_Fails column A
            cashFailsSheet.Cells(i, "A").Value = inventorySheet.Cells(matchResult + 1, "H").Value ' +1 to account for header row
        Else
            ' If no match is found, you can handle it as needed
            cashFailsSheet.Cells(i, "A").Value = "No Match"
        End If
    Next i
End Sub
