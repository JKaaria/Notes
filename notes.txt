Sub CreateOverviewSheet()
    ' Create Overview sheet
    Sheets.Add(After:=Sheets(Sheets.Count)).Name = "Overview"

    ' Summarize key metrics
    Sheets("Overview").Range("A1").Value = "Total Value of Offerings"
    Sheets("Overview").Range("B1").Formula = "=SUM('All Offerings'!B2:B100)"  ' Update the range as needed

    ' Average Coupon
    Sheets("Overview").Range("A2").Value = "Average Coupon"
    Sheets("Overview").Range("B2").Formula = "=AVERAGE('All Offerings'!R2:R100)"  ' Update the range as needed
End Sub

Sub CreateSectorAnalysisSheet()
    ' Create Sector Analysis sheet
    Sheets.Add(After:=Sheets(Sheets.Count)).Name = "Sector Analysis"

    ' Analyze distribution of offerings across sectors
    ' Assume that Sector is in column W in "All Offerings" sheet
    Sheets("Sector Analysis").Range("A1").Value = "Sector"
    Sheets("Sector Analysis").Range("B1").Value = "Number of Offerings"
    Sheets("Sector Analysis").Range("A2").Formula = "=UNIQUE('All Offerings'!W2:W100)"  ' Update the range as needed
    Sheets("Sector Analysis").Range("B2").Formula = "=COUNTIF('All Offerings'!W2:W100,A2)"  ' Update the range as needed

    ' Create a bar chart
    Dim cht As ChartObject
    Set cht = Sheets("Sector Analysis").Shapes.AddChart2(201, xlColumnClustered, 100, 100, 400, 300).Chart
    cht.SetSourceData Source:=Sheets("Sector Analysis").Range("A1:B" & Sheets("Sector Analysis").Cells(Rows.Count, 1).End(xlUp).Row)
End Sub

Sub CreateTimeSeriesAnalysisSheet()
    ' Create Time Series Analysis sheet
    Sheets.Add(After:=Sheets(Sheets.Count)).Name = "Time Series Analysis"

    ' Use a line chart to show how the total value of offerings changes over time
    ' Assume that Offer Date and Settle Date are in columns X and Y in "All Offerings" sheet
    Sheets("Time Series Analysis").Range("A1").Value = "Date"
    Sheets("Time Series Analysis").Range("B1").Value = "Total Value"
    Sheets("Time Series Analysis").Range("A2").Formula = "=UNIQUE('All Offerings'!X2:X100)"  ' Update the range as needed
    Sheets("Time Series Analysis").Range("B2").Formula = "=SUMIFS('All Offerings'!B2:B100, 'All Offerings'!X2:X100, A2)"  ' Update the range as needed

    ' Create a line chart
    Dim cht As ChartObject
    Set cht = Sheets("Time Series Analysis").Shapes.AddChart2(201, xlLine, 100, 100, 400, 300).Chart
    cht.SetSourceData Source:=Sheets("Time Series Analysis").Range("A1:B" & Sheets("Time Series Analysis").Cells(Rows.Count, 1).End(xlUp).Row)
End Sub

Sub CreateDashboardNavigation()
    ' Create hyperlinks on each sheet for easy navigation
    For Each ws In Sheets
        If ws.Name <> "DashboardNavigation" Then
            Sheets("DashboardNavigation").Hyperlinks.Add _
                Anchor:=Sheets("DashboardNavigation").Cells(Sheets("DashboardNavigation").Cells(Rows.Count, 1).End(xlUp).Row + 1, 1), _
                Address:="", SubAddress:="'" & ws.Name & "'!A1", TextToDisplay:=ws.Name
        End If
    Next ws
End Sub
