Sub CleanAndFormatData()
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim col As Range

    ' Set the worksheet
    Set ws = ThisWorkbook.Sheets("All Offerings")

    ' Apply filters to all columns
    ws.Range("A1").AutoFilter Field:=1, Criteria1:="<>"

    ' Find the last row with data
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

    ' Loop through each column and fill down empty cells
    For Each col In ws.UsedRange.Columns
        col.SpecialCells(xlCellTypeBlanks).Formula = "=IF(R[-1]C<>"", R[-1]C, """")"
        col.Value = col.Value
    Next col

    ' Remove filters
    ws.AutoFilterMode = False

    ' Autofit columns for better visibility
    ws.Columns.AutoFit

    MsgBox "Data has been cleaned and formatted successfully!", vbInformation
End Sub
