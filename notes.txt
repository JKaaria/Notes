Sub Three()
    Dim wsTrade As Worksheet
    Dim LastRow As Long
    Dim QuantityCol As Integer
    Dim i As Long

    ' Set worksheet
    Set wsTrade = ThisWorkbook.Sheets("TradeData")
    
    ' Set column number based on header "Quantity"
    QuantityCol = WorksheetFunction.Match("Quantity", wsTrade.Rows(1), 0)
    
    ' Find the last row in the sheet
    LastRow = wsTrade.Cells(wsTrade.Rows.Count, QuantityCol).End(xlUp).Row
    
    ' Loop through "Quantity" column starting from the second row
    For i = LastRow To 2 Step -1
        ' Check if the value is under 100,000,000
        If wsTrade.Cells(i, QuantityCol).Value < 100000000 Then
            ' If so, remove the row
            wsTrade.Rows(i).Delete
        End If
    Next i
End Sub
