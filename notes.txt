Sub ConvertTextToNumbers()
    Dim wsAllOfferings As Worksheet
    Dim dataColumns As Variant
    Dim cell As Range

    ' Set references to the "All Offerings" sheet
    On Error Resume Next
    Set wsAllOfferings = ThisWorkbook.Sheets("All Offerings")
    On Error GoTo 0

    If wsAllOfferings Is Nothing Then
        MsgBox "Sheet 'All Offerings' not found!", vbExclamation
        Exit Sub
    End If

    ' Columns to convert to decimals (adjust as needed)
    dataColumns = Array(3, 13, 16, 19, 20, 26, 27, 28)

    ' Disable alerts to prevent unnecessary prompts
    Application.DisplayAlerts = False

    ' Loop through each column specified in dataColumns
    For Each col In dataColumns
        ' Find the last row in the column
        lastRow = wsAllOfferings.Cells(wsAllOfferings.Rows.Count, col).End(xlUp).Row

        ' Loop through each cell in the column
        For row = 2 To lastRow
            ' Check if the cell contains text that can be converted to a number
            If IsNumeric(wsAllOfferings.Cells(row, col).Value) Then
                ' Convert the cell value to a decimal
                wsAllOfferings.Cells(row, col).Value = CDbl(wsAllOfferings.Cells(row, col).Value) / 100
            End If
        Next row
    Next col

    ' Re-enable alerts
    Application.DisplayAlerts = True

    MsgBox "Conversion completed successfully!", vbInformation
End Sub
