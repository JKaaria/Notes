Function GetRecipientEmail(logicTable As Worksheet, value1 As Variant, value2 As Variant) As String
    Dim xRange As Range
    Dim yRange As Range
    Dim intersectRange As Range
    Dim recipientEmail As String
    
    ' Define ranges for x and y axis
    Set xRange = logicTable.Rows(1).Find(What:=value1, LookIn:=xlValues, LookAt:=xlWhole)
    Set yRange = logicTable.Columns(1).Find(What:=value2, LookIn:=xlValues, LookAt:=xlWhole)
    
    ' Check if both x and y values are found
    If Not xRange Is Nothing And Not yRange Is Nothing Then
        ' Find the intersection of x and y values
        Set intersectRange = logicTable.Cells(yRange.Row, xRange.Column)
        
        ' Check if intersection value is not empty
        If Not intersectRange Is Nothing Then
            ' Return the email address from the intersection cell
            recipientEmail = intersectRange.Value
        End If
    End If
    
    ' Return the email address
    GetRecipientEmail = recipientEmail
End Function
