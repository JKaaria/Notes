Sub DisplayColumnNames()
    Dim wsData As Worksheet
    Set wsData = ThisWorkbook.Sheets("All Offerings")

    If wsData Is Nothing Then
        MsgBox "Sheet 'All Offerings' not found!", vbExclamation
        Exit Sub
    End If

    On Error Resume Next
    Dim tbl As ListObject
    Set tbl = wsData.ListObjects(1)
    On Error GoTo 0

    If tbl Is Nothing Then
        MsgBox "No structured table (ListObject) found in 'All Offerings' sheet.", vbExclamation
        Exit Sub
    End If

    Dim col As ListColumn
    For Each col In tbl.ListColumns
        Debug.Print col.Name
    Next col
End Sub
