Function GetRecipientEmail(logicTable As Worksheet, value1 As Variant, value2 As Variant) As String
    Dim lastRow As Long
    Dim foundRowX As Range
    Dim foundRowY As Range
    Dim recipientEmail As String
    
    ' Find matching row based on value1 in LogicTable
    lastRow = logicTable.Cells(logicTable.Rows.Count, 1).End(xlUp).Row
    Set foundRowX = logicTable.Range("A1:A" & lastRow).Find(What:=value1)
    
    ' Check if a match is found
    If Not foundRowX Is Nothing Then
        ' Find matching row based on value2 in LogicTable
        Set foundRowY = logicTable.Rows(1).Find(What:=value2, LookIn:=xlValues, LookAt:=xlWhole)
        
        ' Check if a match is found
        If Not foundRowY Is Nothing Then
            ' Return the email address from the intersection of the respective x/y axis
            recipientEmail = logicTable.Cells(foundRowX.Row, foundRowY.Column).Value
        End If
    End If
    
    GetRecipientEmail = recipientEmail
End Function
