Sub DisplayColumnNames()
    Dim wsData As Worksheet
    Set wsData = ThisWorkbook.Sheets("All Offerings")

    If wsData Is Nothing Then
        MsgBox "Sheet 'All Offerings' not found!", vbExclamation
        Exit Sub
    End If

    Dim lastCol As Long
    lastCol = wsData.Cells(1, wsData.Columns.Count).End(xlToLeft).Column

    Dim col As Range
    For Each col In wsData.Range(wsData.Cells(1, 1), wsData.Cells(1, lastCol))
        Debug.Print col.Value
    Next col
End Sub
