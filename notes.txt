Sub Three()
    Dim wsTradeData As Worksheet
    Dim LastRow As Long
    Dim issuerAbbrevCol As Integer
    Dim i As Long
    
    ' Set worksheet
    Set wsTradeData = ThisWorkbook.Sheets("TradeData")
    
    ' Set column number based on header "Issuer Abbrev"
    issuerAbbrevCol = WorksheetFunction.Match("Issuer Abbrev", wsTradeData.Rows(1), 0)
    
    ' Find the last row in the sheet
    LastRow = wsTradeData.Cells(wsTradeData.Rows.Count, issuerAbbrevCol).End(xlUp).Row
    
    ' Loop through "Issuer Abbrev" column starting from the second row
    For i = LastRow To 2 Step -1
        ' Check if the value contains "US Treasury WI"
        If InStr(1, wsTradeData.Cells(i, issuerAbbrevCol).Value, "US Treasury WI") > 0 Then
            ' If yes, remove the row
            wsTradeData.Rows(i).Delete
        End If
    Next i
End Sub
