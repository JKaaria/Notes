Sub ResolveTypeMismatches()
    Dim ws As Worksheet
    Dim rng As Range
    Dim cell As Range

    ' Assuming your data is in the "All Offerings" sheet and starts from column B (2) to column AV (49)
    Set ws = ThisWorkbook.Sheets("All Offerings")
    Set rng = ws.Range("B2:AV10000")  ' Update the range as needed

    For Each cell In rng
        ' Check if the cell is not empty
        If Not IsEmpty(cell.Value) Then
            ' Attempt to convert the value to Double (you can modify as needed)
            On Error Resume Next
            cell.Value = CDbl(cell.Value)
            On Error GoTo 0
        End If
    Next cell
End Sub
