Sub FilterSalesPersonNames()

    ' Define variables
    Dim wsTrade As Worksheet
    Dim rngSalesPerson As Range
    Dim salesPersonHeader As Range
    Dim filterNames As Variant
    Dim cell As Range
    
    ' Set worksheet
    Set wsTrade = ThisWorkbook.Sheets("TradeData") ' Replace "TradeData" with the actual name of your trade data sheet
    
    ' Find "Sales Person" column header
    On Error Resume Next
    Set salesPersonHeader = wsTrade.Rows(1).Find("Sales Person", LookIn:=xlValues)
    On Error GoTo 0
    
    ' Check if header is found
    If Not salesPersonHeader Is Nothing Then
        ' Set range for "Sales Person" column
        Set rngSalesPerson = wsTrade.Columns(salesPersonHeader.Column)
        
        ' Define array of names to filter
        filterNames = Array("Devi Aujeet", "David Skingle", "Rui Leal", "Michael Laosun", _
                            "Justin Haskins", "Ksanet Yohannes", "Shahil Pattni", _
                            "Malvi Ruparelia", "Edgar Rodriguez")
        
        ' Loop through each cell in the "Sales Person" column
        For Each cell In rngSalesPerson.Cells
            ' Check if the cell value contains any of the specified names
            If Not IsError(Application.Match(cell.Value, filterNames, 0)) Then
                ' If a match is found, display a message (you can modify this part as needed)
                MsgBox "Found: " & cell.Value, vbInformation
            End If
        Next cell
    Else
        MsgBox "Column header 'Sales Person' not found!", vbExclamation
    End If

End Sub
