Sub InsertColumnAndUpdateValues()
    Dim cashFailsSheet As Worksheet
    Dim inventorySheet As Worksheet
    Dim lastRowCashFails As Long
    Dim cashFailsRange As Range
    Dim cell As Range
    Dim lookupValue As Variant
    Dim foundCell As Range

    ' Set references to the sheets
    Set cashFailsSheet = ThisWorkbook.Sheets("FI_Cash_Fails")
    Set inventorySheet = ThisWorkbook.Sheets("FI_FISS_Inventory")

    ' Find the last row in "FI_Cash_Fails" sheet
    lastRowCashFails = cashFailsSheet.Cells(cashFailsSheet.Rows.Count, "U").End(xlUp).Row

    ' Set the range for the lookup column in "FI_Cash_Fails" sheet
    Set cashFailsRange = cashFailsSheet.Range("U2:U" & lastRowCashFails)

    ' Loop through each cell in the lookup range
    For Each cell In cashFailsRange
        ' Get the value to look up
        lookupValue = cell.Value

        ' Search for the value in "FI_FISS_Inventory" sheet column D
        Set foundCell = inventorySheet.Columns("D").Find(lookupValue, LookIn:=xlValues, LookAt:=xlWhole)

        ' If a match is found, update the corresponding value in "FI_Cash_Fails" column A
        If Not foundCell Is Nothing Then
            cell.Offset(0, -20).Value = foundCell.Offset(0, 4).Value
        End If
    Next cell
End Sub
