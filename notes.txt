Sub ResolveTypeMismatches()
    Dim myString As String
    Dim myNumber As Double
    Dim myInteger As Integer
    Dim myBoolean As Boolean

    ' Type mismatch example 1: Assigning a string to a numeric variable
    On Error Resume Next
    myNumber = "123"
    On Error GoTo 0
    If Err.Number <> 0 Then
        Debug.Print "Type mismatch: Unable to assign a string to a numeric variable."
        Err.Clear
    Else
        Debug.Print "No type mismatch."
    End If

    ' Type mismatch example 2: Assigning a double to an integer variable
    On Error Resume Next
    myInteger = 123.45
    On Error GoTo 0
    If Err.Number <> 0 Then
        Debug.Print "Type mismatch: Unable to assign a double to an integer variable."
        Err.Clear
    Else
        Debug.Print "No type mismatch."
    End If

    ' Type mismatch example 3: Using a boolean variable in a numeric context
    On Error Resume Next
    myBoolean = True
    myNumber = myNumber + myBoolean
    On Error GoTo 0
    If Err.Number <> 0 Then
        Debug.Print "Type mismatch: Unable to use a boolean variable in a numeric context."
        Err.Clear
    Else
        Debug.Print "No type mismatch."
    End If
End Sub
