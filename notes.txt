Sub RenameSourceFile()

    ' Define variables
    Dim sourcePath As String
    Dim sourceFileName As String
    
    ' Set source file path to the current workbook's folder
    sourcePath = ThisWorkbook.Path & "\"
    
    ' Find the file in the current workbook's folder that is not named "RepoSalesFICashTrades.xlsm"
    sourceFileName = Dir(sourcePath & "*.xlsm") ' Update the file extension as needed
    
    ' Check if a valid source file is found
    If sourceFileName <> "" And sourceFileName <> "RepoSalesFICashTrades.xlsm" Then
        ' Rename the source file to "SourceFile.xls"
        Name sourcePath & sourceFileName As sourcePath & "SourceFile.xls"
        MsgBox "File renamed successfully!", vbInformation
    Else
        MsgBox "No valid source file found in the current workbook's folder or file is already named 'RepoSalesFICashTrades.xlsm'.", vbExclamation
    End If

End Sub
