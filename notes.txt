Sub ConvertRowToColumn()
    Dim ws As Worksheet
    Dim dataRange As Range
    Dim resultRange As Range
    Dim lastCol As Long
    Dim lastRow As Long
    
    ' Set the worksheet (change Sheet1 to your actual sheet name)
    Set ws = ThisWorkbook.Sheets("Sheet1")
    
    ' Find the last column with data in the first row
    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
    
    ' Set the data range (adjust the range as needed)
    Set dataRange = ws.Range("A1:" & ws.Cells(1, lastCol).Address)
    
    ' Find the last row in the data range
    lastRow = dataRange.Rows.Count
    
    ' Set the result range (starting from cell B1, adjust as needed)
    Set resultRange = ws.Range("B1").Resize(lastRow, 1)
    
    ' Copy and transpose the data
    dataRange.Copy
    resultRange.PasteSpecial Paste:=xlPasteAll, Operation:=xlNone, SkipBlanks:=False, Transpose:=True
    Application.CutCopyMode = False
End Sub
