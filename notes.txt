Sub RenameAndCopyFiles()
    Dim DataFolder As String
    Dim File1 As String
    Dim File2 As String
    Dim SourceWorkbook As Workbook
    
    ' Define the path to the "data" folder
    DataFolder = ThisWorkbook.Path & "\dashboard\data\"
    
    ' Check if the "data" folder exists
    If Dir(DataFolder, vbDirectory) <> "" Then
        ' Get the list of files in the "data" folder
        File1 = Dir(DataFolder & "*.*")
        If File1 <> "" Then
            ' Rename the first file to "delphi"
            Name DataFolder & File1 As DataFolder & "delphi"
            ' Copy data from "delphi" to "Delphi" sheet in Workbook1
            Set SourceWorkbook = Workbooks.Open(DataFolder & "delphi")
            SourceWorkbook.Sheets(1).UsedRange.Copy ThisWorkbook.Sheets("Delphi").Cells(1, 3)
            SourceWorkbook.Close SaveChanges:=False
        End If
        
        File2 = Dir(DataFolder & "*.*")
        If File2 <> "" Then
            ' Rename the second file to "fiss"
            Name DataFolder & File2 As DataFolder & "fiss"
            ' Copy data from "fiss" to "FISS" sheet in Workbook1
            Set SourceWorkbook = Workbooks.Open(DataFolder & "fiss")
            SourceWorkbook.Sheets(1).UsedRange.Copy ThisWorkbook.Sheets("FISS").Cells(1, 1)
            SourceWorkbook.Close SaveChanges:=False
        End If
    Else
        MsgBox "The 'data' folder does not exist in the same directory as this workbook.", vbExclamation
    End If
End Sub
