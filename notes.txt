Sub CreateOverviewSheet()
    ' Declare variables
    Dim wsAllOfferings As Worksheet
    Dim wsOverview As Worksheet
    Dim lastRow As Long
    Dim i As Long
    
    ' Set references to worksheets
    Set wsAllOfferings = ThisWorkbook.Sheets("All Offerings")
    
    ' Check if "Overview" sheet already exists, if yes, delete it
    For Each ws In ThisWorkbook.Sheets
        If ws.Name = "Overview" Then
            Application.DisplayAlerts = False
            ws.Delete
            Application.DisplayAlerts = True
            Exit For
        End If
    Next ws
    
    ' Create a new sheet named "Overview"
    Set wsOverview = ThisWorkbook.Sheets.Add(After:=Sheets(Sheets.Count))
    wsOverview.Name = "Overview"
    
    ' Set headers in the "Overview" sheet
    wsOverview.Range("A1").Value = "Issuer"
    wsOverview.Range("B1").Value = "Total Offer Amount (MM)"
    wsOverview.Range("C1").Value = "Average Price (Discount)"
    ' Add more headers as needed
    
    ' Find the last row in the "All Offerings" sheet
    lastRow = wsAllOfferings.Cells(wsAllOfferings.Rows.Count, "A").End(xlUp).Row
    
    ' Loop through the data in "All Offerings" and summarize key metrics
    For i = 2 To lastRow ' Assuming data starts from row 2 (header in row 1)
        ' Example: Summarize issuer-wise total offer amount
        wsOverview.Cells(i, 1).Value = wsAllOfferings.Cells(i, 1).Value ' Copy Issuer
        wsOverview.Cells(i, 2).Value = Application.WorksheetFunction.SumIf(wsAllOfferings.Range("A:A"), wsAllOfferings.Cells(i, 1).Value, wsAllOfferings.Range("B:B")) ' Sum Total Offer Amount for the Issuer
        wsOverview.Cells(i, 3).Value = Application.WorksheetFunction.AverageIf(wsAllOfferings.Range("A:A"), wsAllOfferings.Cells(i, 1).Value, wsAllOfferings.Range("C:C")) ' Average Price (Discount) for the Issuer
        ' Add more summarization as needed
    Next i
End Sub
