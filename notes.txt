# Import the UIAutomation module
Import-Module UIAutomation

# Start the target application
Start-Process -FilePath "C:\Path\To\Your\Application.exe" -PassThru | Wait-Process

# Find the main window of the application
$mainWindow = Get-UIAWindow -Name "Your Application Name"

# Find the "to Excel" button
$excelButton = $mainWindow | Get-UIAButton -Name "To Excel"

# Click the "to Excel" button
$excelButton | Invoke-UIAButtonClick
